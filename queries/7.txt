select *  from tags
join posts ps 
on ps.id = tags.excerptpostid

where (extract (year from ps.creationdate) = 2014 )
order by tags.count asc
having ps.posttypeid = 2

join posts ps 
on ps.id = tags.excerptpostid

where (extract (year from ps.creationdate) = 2014 )
order by tags.count asc
